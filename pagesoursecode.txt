    function getNotificationList() {

        $.ajax({
            type:"POST",
            url:"/academia/userNotification/notificationList?pageNo=0",
            beforeSend:function (jqXHR, settings) {
                $("#loader_icon").show();
            },
            success:function (entity) {
                var count = 0;
                $.each(entity.userNotificationList, function(key,value) {
                    if(value.message_status == false){
                        $("#test-login-container-new-message ul").prepend('<li><a href="#/academia/userNotification/showAll?id='+value.id+'"><span class="tab">'+ value.subject +'</span></a></li>');
                        count++
                    }

                });
                $('#notificationCount').text(entity.totalUnread);

//                    $("#notificationUI ul").append('<li><a href="/user/messages"><span class="tab">Message Center</span></a></li>');
            },
            complete:function () {
                $("#loader_icon").hide();
            },
            dataType:'json'
        });
    }
